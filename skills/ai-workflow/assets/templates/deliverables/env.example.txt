# =============================================================================
# .env.example  —  template, safe to commit
# -----------------------------------------------------------------------------
# Copy to .env and fill in REAL values locally:  cp .env.example .env
# The real `.env` is gitignored and MUST NOT be committed.
# Real secrets live in 1Password / Vault — never paste production secrets here.
# Values below are PLACEHOLDERS only.
#
# NOTE: The ACTUAL keys depend on the stack and providers chosen in
# Stack + Instrumentation — database engine, auth approach, storage/email/
# analytics/payments providers, and whether the runtime exposes client-side
# vars under a framework prefix at all. The groups below are a checklist of
# *categories* to fill per project, not a set of defaults to copy verbatim.
# Delete groups you don't use; rename keys to match your providers.
# =============================================================================

# ---- App ----
# APP_ENV / NODE_ENV / RAILS_ENV / etc. — match your runtime's convention.
APP_ENV=development
APP_URL=http://localhost:<port>
APP_NAME=your-app-name
LOG_LEVEL=info

# ---- Database ----
# One connection string per the engine you chose (Postgres, MySQL, SQLite,
# Mongo, etc.). Pick the scheme that matches; this is not Postgres-only.
DATABASE_URL=<scheme>://user:password@localhost:<port>/your-db-name
# DATABASE_POOL_MAX=10

# ---- Auth ----
# Keys depend on the auth approach decided in Stack (managed service,
# self-hosted, session vs token, which OAuth/SSO providers — if any).
AUTH_SECRET=your-secret-here
AUTH_URL=http://localhost:<port>
# Per-provider OAuth/SSO credentials, only if you use them:
# <PROVIDER>_CLIENT_ID=your-client-id-here
# <PROVIDER>_CLIENT_SECRET=your-secret-here

# ---- Storage ----
# Only if the project stores files. Fields vary by provider (S3-compatible,
# GCS, Azure Blob, local disk…). Add region/endpoint only if yours needs it.
STORAGE_BUCKET=your-bucket-name
STORAGE_ACCESS_KEY_ID=your-key-here
STORAGE_SECRET_ACCESS_KEY=your-secret-here
# STORAGE_REGION=your-region
# STORAGE_ENDPOINT=your-endpoint

# ---- Email ----
# Only if the project sends mail. Key shape depends on the provider chosen.
EMAIL_API_KEY=your-key-here
EMAIL_FROM=noreply@your-domain.com

# ---- Analytics ----
# Only if you track analytics. If your framework exposes client-side vars,
# they carry a framework-specific prefix (e.g. NEXT_PUBLIC_ / VITE_ / PUBLIC_)
# — use whatever your stack actually requires, if any.
ANALYTICS_WRITE_KEY=your-key-here
# <PUBLIC_PREFIX>_ANALYTICS_ID=your-public-id-here

# ---- Payments ----
# Only if the project takes payments. Secret keys stay server-side; any
# publishable/client key uses your framework's public prefix, if it has one.
PAYMENTS_SECRET_KEY=your-secret-here
PAYMENTS_WEBHOOK_SECRET=your-secret-here
# <PUBLIC_PREFIX>_PAYMENTS_PUBLISHABLE_KEY=your-public-key-here
